function sendImageMessage() {
  // Added detailed logging
  console.log('[AIDP Chat] Socket state:', socket ? 'exists' : 'null');
  console.log('[AIDP Chat] Recipient ID:', recipientId);

  // Check socket exists
  if (!socket) {
    alert('Chat connection not initialized. Please refresh the page.');
    return;
  }

  // Check socket connected with retry
  if (!socket.connected) {
    setTimeout(() => {
      if (socket && socket.connected) {
        sendImageMessage(); // Retry after 1 second
      } else {
        alert('Connection lost. Please refresh the page.');
      }
    }, 1000);
    return;
  }

  // Check recipient ID (including string 'null')
  if (!recipientId || recipientId === 'null' || recipientId === null) {
    alert('Error: No recipient selected. Please select a user first.');
    return;
  }

  // Send image...
}
